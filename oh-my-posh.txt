function Get-WifiPassword {
    [CmdletBinding()]
    param (
        [Parameter(Mandatory=$true, Position=0)]
        [string]$SSID
    )

    $profiles = netsh wlan show profiles
    $profileNames = $profiles | Select-String "All User Profile" | ForEach-Object { $_.ToString().Split(":")[1].Trim() }

    if ($profileNames -contains $SSID) {
        $password = (netsh wlan show profile name="$SSID" key=clear) -match "Key Content\s+:\s+(.*)" | Out-Null
        Write-Output "Wi-Fi SSID: $SSID"
        Write-Output "Password: $($Matches[1])"
    }
    else {
        Write-Output "Wi-Fi SSID '$SSID' not found."
    }
}
